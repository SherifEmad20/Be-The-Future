(BACKEND)
zookeeper server start:
.\bin\windows\zookeeper-server-start.bat .\config\zookeeper.properties

kafka server start:
.\bin\windows\kafka-server-start.bat .\config\server.properties

topic create:
.\bin\windows\kafka-topics.bat --create --topic kafka-chat-2 --bootstrap-server localhost:9092 --replication-factor 1 --partitions 1

topic consumer:
.\bin\windows\kafka-console-consumer.bat --bootstrap-server localhost:9092 --topic kafka-chat-2
